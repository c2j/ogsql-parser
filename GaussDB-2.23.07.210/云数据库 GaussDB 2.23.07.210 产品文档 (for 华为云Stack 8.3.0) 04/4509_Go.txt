func main() {
    hostip := os.Getenv("GOHOSTIP")   //GOHOSTIP为写入环境变量的IP地址

    passwd := os.Getenv("GOPASSWD")   //GOPASSWDW为写入环境变量的用户密码
    str := "host=" + hostip + " port=" + port + " user=" + usrname + " password=" + passwd + " dbname=postgres enable_ce=3" // DSN连接串
    // str := "opengauss://" + usrname + ":" + passwd + "@" + hostip + ":" + port + "/postgres?enable_ce=3"  // URL连接串

    // 获取数据库连接池句柄
    db, err := sql.Open("opengauss", str)
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    // Open函数仅是验证参数，Ping方法可检查数据源是否合法
    err = db.Ping()
    if err == nil {
        fmt.Printf("Connection succeed!\n")

        log.Fatal(err)
    }
